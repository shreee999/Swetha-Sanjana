import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.preprocessing import StandardScaler
from sklearn.impute import SimpleImputer

# -----------------------------
# Step 0: Load dataset
# -----------------------------
# df = pd.read_csv("steam_games.csv")  # Uncomment if loading from CSV

# -----------------------------
# Step 1: Select features for value analysis
# -----------------------------
# We want measurable value vs price
numeric_features = ['Positive', 'Negative', 'User score', 'Metacritic score',
                    'Estimated owners', 'Peak CCU', 'Recommendations',
                    'Achievements', 'Required age', 'Price', 'DiscountDLC count',
                    'total_reviews', 'review_score']

# -----------------------------
# Step 2: Handle missing values
# -----------------------------
imputer = SimpleImputer(strategy='mean')
features_clean = imputer.fit_transform(df[numeric_features])

# Convert back to DataFrame for convenience
features_df = pd.DataFrame(features_clean, columns=numeric_features)

# -----------------------------
# Step 3: Compute a "Value Score"
# -----------------------------
# Simple approach: Value = Positive reviews / Price
# Add a small epsilon to avoid division by zero
epsilon = 1e-5
features_df['Value_Score'] = features_df['Positive'] / (features_df['Price'] + epsilon)

# Optional: normalize Value_Score for easier interpretation
features_df['Value_Score_Norm'] = (features_df['Value_Score'] - features_df['Value_Score'].min()) / \
                                  (features_df['Value_Score'].max() - features_df['Value_Score'].min())

# Add it to the main DataFrame
df['Value_Score'] = features_df['Value_Score']
df['Value_Score_Norm'] = features_df['Value_Score_Norm']

# -----------------------------
# Step 4: Identify extremes
# -----------------------------
# Top 5 overperformers (high value)
top_overperformers = df.sort_values(by='Value_Score', ascending=False).head(5)

# Top 5 underperformers (low value, price > 0)
top_underperformers = df[df['Price'] > 0].sort_values(by='Value_Score', ascending=True).head(5)

print("Top 5 Overperformers:")
print(top_overperformers[['Name', 'Price', 'Positive', 'Value_Score']])
print("\nTop 5 Underperformers:")
print(top_underperformers[['Name', 'Price', 'Positive', 'Value_Score']])

# -----------------------------
# Step 5: Visualization
# -----------------------------
plt.figure(figsize=(12,6))
sns.scatterplot(
    x='Price', 
    y='Positive', 
    size='Estimated owners',
    hue='Value_Score_Norm',
    data=df,
    palette='viridis',
    alpha=0.7,
    sizes=(20, 200)
)
plt.title("Market Paradox: Price vs Positive Reviews (Size = Estimated Owners, Color = Value Score)")
plt.xlabel("Price ($)")
plt.ylabel("Positive Reviews")
plt.legend([],[], frameon=False)  # optional: hide legend clutter
plt.show()

# -----------------------------
# Step 6: Optional — Highlight extremes
# -----------------------------
plt.figure(figsize=(12,6))
sns.scatterplot(
    x='Price', 
    y='Positive', 
    data=df, 
    alpha=0.3
)
# Highlight overperformers in green
sns.scatterplot(x=top_overperformers['Price'], y=top_overperformers['Positive'],
                color='green', s=150, label='Top Overperformers')
# Highlight underperformers in red
sns.scatterplot(x=top_underperformers['Price'], y=top_underperformers['Positive'],
                color='red', s=150, label='Top Underperformers')
plt.title("Steam Market Paradox: Overperformers vs Underperformers")
plt.xlabel("Price ($)")
plt.ylabel("Positive Reviews")
plt.legend()
plt.show()
