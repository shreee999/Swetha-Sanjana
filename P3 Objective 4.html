import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# -----------------------------
# Step 0: Load dataset
# -----------------------------
# df = pd.read_csv("steam_games.csv")  # Uncomment to load CSV

# -----------------------------
# Step 1: Convert numeric features
# -----------------------------
numeric_features = ['Positive', 'Negative', 'User score', 'Metacritic score',
                    'Estimated owners', 'Peak CCU', 'Recommendations',
                    'Achievements', 'Required age', 'Price', 'DiscountDLC count',
                    'total_reviews', 'review_score']

for col in numeric_features:
    df[col] = pd.to_numeric(df[col], errors='coerce')

# -----------------------------
# Step 2: Convert Name column (which has the date) to Year
# -----------------------------
df['Release Year'] = pd.to_datetime(df['Name'], errors='coerce', format='%b %d, %Y').dt.year

# -----------------------------
# Step 3: Keep rows with valid numeric data and Release Year
# -----------------------------
df_clean = df[df['Release Year'].notna() & df['Positive'].notna()]

print(f"Total rows after cleaning: {len(df_clean)}")

# -----------------------------
# Step 4: Aggregate numeric satisfaction metrics by year
# -----------------------------
yearly_satisfaction = df_clean.groupby('Release Year').agg(
    Avg_Positive=('Positive', 'mean'),
    Avg_User_Score=('User score', 'mean'),
    Num_Games=('Name', 'count')
).reset_index()

# -----------------------------
# Step 5: Plot Average Positive Reviews
# -----------------------------
if not yearly_satisfaction.empty:
    plt.figure(figsize=(12,6))
    sns.lineplot(data=yearly_satisfaction, x='Release Year', y='Avg_Positive', marker='o')
    plt.title("Average Positive Reviews Over Years")
    plt.xlabel("Release Year")
    plt.ylabel("Average Positive Reviews")
    plt.show()

    # -----------------------------
    # Step 6: Plot Average User Score
    # -----------------------------
    plt.figure(figsize=(12,6))
    sns.lineplot(data=yearly_satisfaction, x='Release Year', y='Avg_User_Score', marker='o', color='orange')
    plt.title("Average User Score Over Years")
    plt.xlabel("Release Year")
    plt.ylabel("Average User Score")
    plt.show()

    # -----------------------------
    # Step 7: Plot Number of Games Released Per Year
    # -----------------------------
    plt.figure(figsize=(12,6))
    sns.barplot(data=yearly_satisfaction, x='Release Year', y='Num_Games', color='teal')
    plt.title("Number of Games Released Each Year")
    plt.xlabel("Release Year")
    plt.ylabel("Number of Games")
    plt.xticks(rotation=45)
    plt.show()
else:
    print("No valid data available to plot.")
