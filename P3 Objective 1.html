import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# -----------------------------
# Step 0: Load dataset
# -----------------------------
# df = pd.read_csv("steam_games.csv")  # Uncomment if loading CSV

# -----------------------------
# Step 1: Convert numeric columns
# -----------------------------
numeric_features = ['Positive', 'Negative', 'User score', 'Metacritic score',
                    'Estimated owners', 'Peak CCU', 'Reviews', 'Recommendations',
                    'Achievements', 'Required age', 'Price', 'DiscountDLC count',
                    'total_reviews', 'review_score']

for col in numeric_features:
    df[col] = pd.to_numeric(df[col], errors='coerce')

# -----------------------------
# Step 2: Filter games with enough reviews
# -----------------------------
# Avoid games with very few reviews (no meaningful stats)
df_clean = df[df['total_reviews'].notna() & (df['total_reviews'] > 50)]
print(f"Total games considered: {len(df_clean)}")

# -----------------------------
# Step 3: Calculate "Love-Hate Ratio"
# -----------------------------
# Ratio of Positive / Negative reviews
df_clean['Love_Hate_Ratio'] = df_clean['Positive'] / (df_clean['Negative'] + 1)  # +1 to avoid division by zero

# -----------------------------
# Step 4: Identify extremes
# -----------------------------
# Universally adored: high positive, low negative
universally_adored = df_clean.sort_values(['Love_Hate_Ratio', 'total_reviews'], ascending=[False, False]).head(10)

# Highly divided: large number of positive AND negative reviews (polarizing)
polarizing = df_clean.assign(Total_PosNeg=df_clean['Positive'] + df_clean['Negative'])
polarizing = polarizing.sort_values(['Total_PosNeg', 'Love_Hate_Ratio'], ascending=[False, True]).head(10)

# -----------------------------
# Step 5: Plot Love-Hate Divide
# -----------------------------
plt.figure(figsize=(12,6))
sns.scatterplot(data=df_clean, x='Negative', y='Positive', size='total_reviews', hue='Love_Hate_Ratio', palette='coolwarm', alpha=0.7, sizes=(20, 200))
plt.title("Love-Hate Divide: Positive vs Negative Reviews")
plt.xlabel("Negative Reviews")
plt.ylabel("Positive Reviews")
plt.legend(title='Love-Hate Ratio', bbox_to_anchor=(1,1))
plt.show()

# -----------------------------
# Step 6: Top Universally Adored Games
# -----------------------------
plt.figure(figsize=(10,6))
sns.barplot(x='Positive', y='Name', data=universally_adored, palette='Greens_r')
plt.title("Top Universally Adored Games")
plt.xlabel("Positive Reviews")
plt.ylabel("Game")
plt.show()

# -----------------------------
# Step 7: Top Polarizing Games
# -----------------------------
plt.figure(figsize=(10,6))
sns.barplot(x='Total_PosNeg', y='Name', data=polarizing, palette='coolwarm')
plt.title("Top Polarizing Games")
plt.xlabel("Total Reviews (Positive + Negative)")
plt.ylabel("Game")
plt.show()
